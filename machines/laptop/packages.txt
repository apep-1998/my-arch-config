# Laptop-specific packages — Intel Iris Xe GPU + power management

# ── Intel GPU (Raptor Lake / Iris Xe) ─────────────────────────────────────────
# Mesa OpenGL implementation (handles Intel GPU rendering)
mesa
# 32-bit Mesa OpenGL (needed for Steam and Wine)
lib32-mesa
# Vulkan driver for Intel integrated GPUs
vulkan-intel
# hardware video decode and encode via VA-API (Intel Quick Sync)
intel-media-driver

# ── Power management ──────────────────────────────────────────────────────────
# advanced power management for Linux laptops (extends battery life)
tlp
# TLP radio device wizard — manages Wi-Fi and Bluetooth on suspend/resume
tlp-rdw
# analyzes power consumption per device and suggests optimizations
powertop
